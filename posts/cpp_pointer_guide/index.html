🎯 C/C++ 指標完全指南：20個致命錯誤與解決方案

1. 空指標解引用
#

1.1 問題描述
# 

// ❌ 錯誤示範
int* p = NULL;
*p = 1;    // 程式崩潰
printf("%d", *p);  // 非法記憶體存取

1.2 正確做法
# 

int* p = NULL;
if (p != NULL) {
    *p = 1;
    printf("%d", *p);
}

1.3 進階解決方案
# 

// 使用智慧指標
std::unique_ptr<int> p = std::make_unique<int>(1);
if (p) {
    std::cout << *p;
}

2. 野指標使用
#

2.1 問題描述
# 

// ❌ 錯誤示範
int* p;           // 未初始化
printf("%d", *p); // 存取未知記憶體位置

2.2 正確做法
# 

int* p = nullptr; // C++11 建議使用 nullptr
// 或
int value = 42;
int* p = &value;  // 指向有效變數

2.3 最佳實踐
# 

class SafePointer {
private:
    int* ptr;
public:
    SafePointer() : ptr(nullptr) {}
    explicit SafePointer(int value) : ptr(new int(value)) {}
    ~SafePointer() { delete ptr; }
    
    bool isValid() const { return ptr != nullptr; }
    int getValue() const { 
        if (!isValid()) throw std::runtime_error("Null pointer");
        return *ptr;
    }
};

3. 記憶體洩漏
#

3.1 問題描述
# 

// ❌ 錯誤示範
void memoryLeak() {
    int* p = new int(42);
    // 忘記 delete
}  // 記憶體洩漏

3.2 正確做法
# 

void noLeak() {
    int* p = new int(42);
    // 使用 p
    delete p;
    p = nullptr;
}

3.3 現代解決方案
# 

void modernWay() {
    // 使用智慧指標
    std::unique_ptr<int> p = std::make_unique<int>(42);
    // 自動管理記憶體
}

4. 重複釋放
#

4.1 問題描述
# 

// ❌ 錯誤示範
int* p = new int(42);
delete p;
delete p;  // 重複釋放，未定義行為

4.2 正確做法
# 

int* p = new int(42);
delete p;
p = nullptr;  // 設為 nullptr 避免重複釋放

4.3 智慧指標解決方案
# 

std::shared_ptr<int> p = std::make_shared<int>(42);
// 自動處理釋放，不需要手動管理

5. 指標越界
#

5.1 陣列越界問題
# 

// ❌ 錯誤示範
int arr[5] = {1, 2, 3, 4, 5};
int* p = arr;
p[5] = 6;    // 越界寫入
printf("%d", p[10]);  // 越界讀取

5.2 正確做法
# 

// 使用 STL 容器
std::vector<int> arr = {1, 2, 3, 4, 5};
if (arr.size() > 5) {
    arr[5] = 6;
}

// 或使用範圍檢查
template<size_t N>
class SafeArray {
    int data[N];
public:
    int& operator[](size_t i) {
        if (i >= N) throw std::out_of_range("Index out of bounds");
        return data[i];
    }
};

6. 懸掛指標
#

6.1 問題描述
# 

// ❌ 錯誤示範
int* getDangling() {
    int local = 42;
    return &local;  // 回傳區域變數位址
}

// 或
int* p = new int(42);
int* q = p;
delete p;    // p 被釋放
*q = 100;    // q 成為懸掛指標

6.2 正確做法
# 

// 使用智慧指標
std::shared_ptr<int> getNumber() {
    return std::make_shared<int>(42);
}

// 或
std::shared_ptr<int> p = std::make_shared<int>(42);
std::shared_ptr<int> q = p;  // 共享所有權

7. 指標型別不匹配
#

7.1 問題描述
# 

// ❌ 錯誤示範
int* intPtr = new int(42);
char* charPtr = (char*)intPtr;  // 危險的型別轉換

7.2 正確做法
# 

// 使用 static_cast 或 reinterpret_cast
int* intPtr = new int(42);
// 只在確實需要時使用
char* charPtr = reinterpret_cast<char*>(intPtr);

// 更好的做法是避免型別轉換
template<typename T>
class TypeSafePtr {
    T* ptr;
public:
    explicit TypeSafePtr(T* p) : ptr(p) {}
    T& operator*() { return *ptr; }
};

8. 字串指標錯誤
#

8.1 問題描述
# 

// ❌ 錯誤示範
char* str = "Hello";  // 字串常量不可修改
str[0] = 'h';        // 未定義行為

// 或
char* str = new char[6];
strcpy(str, "Hello");
// 忘記釋放記憶體

8.2 正確做法
# 

// 使用字串陣列
char str[] = "Hello";  // 可修改的拷貝

// 或使用 string
std::string str = "Hello";
str[0] = 'h';  // 安全的修改

// 或使用智慧指標
std::unique_ptr<char[]> str(new char[6]);
strcpy(str.get(), "Hello");

9. 函式指標錯誤
#

9.1 問題描述
# 

// ❌ 錯誤示範
void (*fp)();
fp();  // 呼叫未初始化的函式指標

9.2 正確做法
# 

void func() { std::cout << "Hello"; }

// 基本用法
void (*fp)() = func;
if (fp) fp();

// 現代寫法
std::function<void()> fp = func;
if (fp) fp();

// 或使用 std::bind
auto fp = std::bind(&MyClass::method, &obj);

10. 多執行緒指標問題
#

10.1 問題描述
# 

// ❌ 錯誤示範
class SharedResource {
    int* data;
public:
    void update(int value) {
        *data = value;  // 沒有同步機制
    }
};

10.2 正確做法
# 

class ThreadSafeResource {
    std::shared_ptr<int> data;
    mutable std::mutex mutex;
public:
    void update(int value) {
        std::lock_guard<std::mutex> lock(mutex);
        if (data) *data = value;
    }
    
    int getValue() const {
        std::lock_guard<std::mutex> lock(mutex);
        return data ? *data : 0;
    }
};

11. 循環參考問題
#

11.1 問題描述
# 

// ❌ 錯誤示範
struct Node {
    std::shared_ptr<Node> next;
    std::shared_ptr<Node> prev;
};

void createNodes() {
    auto node1 = std::make_shared<Node>();
    auto node2 = std::make_shared<Node>();
    node1->next = node2;
    node2->prev = node1;  // 造成循環參考
}

11.2 正確做法
# 

struct Node {
    std::shared_ptr<Node> next;
    std::weak_ptr<Node> prev;  // 使用 weak_ptr
};

void createNodes() {
    auto node1 = std::make_shared<Node>();
    auto node2 = std::make_shared<Node>();
    node1->next = node2;
    node2->prev = node1;  // 不會造成循環參考
}

12. 智慧指標誤用

12.1 問題描述
# 

// ❌ 錯誤示範
void processData(std::unique_ptr<Data> data) {
    std::unique_ptr<Data> copy = data;  // 編譯錯誤：unique_ptr 不能複製
}

// 或
std::shared_ptr<int> ptr(new int[10]);  // 錯誤：使用錯誤的刪除器

12.2 正確做法
# 

// 傳遞 unique_ptr
void processData(std::unique_ptr<Data>&& data) {
    // 使用 std::move
    std::unique_ptr<Data> local = std::move(data);
}

// 正確管理陣列
std::shared_ptr<int[]> ptr(new int[10]);
// 或
auto ptr = std::make_unique<int[]>(10);

13. 指標與繼承
#

13.1 問題描述
# 

// ❌ 錯誤示範
class Base { 
public:
    virtual ~Base() = default;
};

class Derived : public Base {
    int* data;
public:
    Derived() : data(new int(42)) {}
    // 忘記實作解構子，造成記憶體洩漏
};

13.2 正確做法
# 

class Base {
public:
    virtual ~Base() = default;
};

class Derived : public Base {
    std::unique_ptr<int> data;
public:
    Derived() : data(std::make_unique<int>(42)) {}
    // 不需要手動實作解構子
};

# 

// ❌ 錯誤示範
std::vector<int*> vec;
for (int i = 0; i < 5; ++i) {
    vec.push_back(new int(i));  // 手動管理記憶體

// 自動清理

std::vector<int> vec;

    vec.push_back(i);
}

15. 指標與例外處理
#

15.1 問題描述
# 

// ❌ 錯誤示範
void riskyFunction() {
    int* p = new int(42);
    // 可能拋出例外
    functionThatMightThrow();
    delete p;  // 如果發生例外，這行不會執行
}

15.2 正確做法
# 

// 使用 RAII
void safeFunction() {
    std::unique_ptr<int> p = std::make_unique<int>(42);
    functionThatMightThrow();  // 即使拋出例外，p 也會被正確釋放
}

// 或使用智慧資源管理
class ResourceGuard {
    FILE* file;
public:
    ResourceGuard(const char* filename) {
        file = fopen(filename, "r");
        if (!file) throw std::runtime_error("Failed to open file");
    }
    
    ~ResourceGuard() {
        if (file) fclose(file);
    }
};

16. 指標與 STL 容器
#

16.1 問題描述
# 

// ❌ 錯誤示範
std::vector<int> vec = {1, 2, 3};
int* p = &vec[0];
vec.push_back(4);  // 可能導致重新配置，p 變成懸掛指標

std::vector<int> vec = {1, 2, 3};

auto it = vec.begin();

int& ref = vec[0];

vec.reserve(4);  // 預先配置空間
int* p = &vec[0];
vec.push_back(4);  // 不會重新配置

17. 自訂刪除器
#

17.1 問題描述
# 

// ❌ 錯誤示範
struct CustomResource {
    void* data;
    CustomResource() : data(specialAlloc()) {}
    ~CustomResource() { 
        free(data);  // 使用錯誤的釋放函式
    }
};

17.2 正確做法
# 

// 自訂刪除器
struct CustomDeleter {
    void operator()(void* p) const {
        specialFree(p);  // 使用正確的釋放函式
    }
};

using CustomPtr = std::unique_ptr<void, CustomDeleter>;

struct CustomResource {
    CustomPtr data;
    CustomResource() : data(specialAlloc(), CustomDeleter()) {}
    // 不需要解構子
};

18. 指標與型別轉換
#

18.1 問題描述
# 

// ❌ 錯誤示範
class Base {};
class Derived : public Base {};

Base* bp = new Derived;
Derived* dp = (Derived*)bp;  // 危險的 C 風格轉換

18.2 正確做法
# 

// 使用 C++ 風格轉換
Base* bp = new Derived;
if (Derived* dp = dynamic_cast<Derived*>(bp)) {
    // 安全的轉換
}

// 使用智慧指標
std::shared_ptr<Base> bp = std::make_shared<Derived>();
if (auto dp = std::dynamic_pointer_cast<Derived>(bp)) {
    // 安全的轉換
}

19. 指標與常數
#

19.1 問題描述
# 

// ❌ 錯誤示範
const int constant = 42;
int* ptr = (int*)&constant;  // 移除 const 限定
*ptr = 43;  // 未定義行為

19.2 正確做法
# 

// 保持 const 正確性
const int constant = 42;
const int* ptr = &constant;
// *ptr = 43;  // 編譯錯誤

// 使用 const 智慧指標
std::shared_ptr<const int> ptr = std::make_shared<const int>(42);

20. 指標與移動語意
#

20.1 問題描述
# 

// ❌ 錯誤示範
class Resource {
    int* data;
public:
    Resource(int value) : data(new int(value)) {}
    ~Resource() { delete data; }
    
    // 缺少移動建構子和賦值運算子
    Resource(const Resource& other) : data(new int(*other.data)) {}
    Resource& operator=(const Resource& other) {
        if (this != &other) {
            delete data;
            data = new int(*other.data);
        }
        return *this;
    }
};

20.2 正確做法
# 

class Resource {
    std::unique_ptr<int> data;
public:
    Resource(int value) : data(std::make_unique<int>(value)) {}
    
    // 移動建構子
    Resource(Resource&& other) noexcept = default;
    
    // 移動賦值運算子
    Resource& operator=(Resource&& other) noexcept = default;
    
    // 禁用拷貝
    Resource(const Resource&) = delete;
    Resource& operator=(const Resource&) = delete;
};

class ThreadSafe {
    mutable std::mutex mutex;

public:

        if (resource) {
            resource->use();
        }
    }
};

#

  1. Valgrind

    • 記憶體洩漏檢測

    • 記憶體存取錯誤檢測

  2. AddressSanitizer (ASAN)

    • 即時記憶體錯誤檢測
    • 效能開銷較小

  3. Visual Studio 除錯器

    • 記憶體檢視
    • 斷點除錯
    • 記憶體洩漏檢測

  4. Static Analysis Tools

    • Clang Static Analyzer
    • Cppcheck
    • PVS-Studio
