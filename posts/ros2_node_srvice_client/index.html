🐢 ROS2 巡邏服務節點開發教學(使用 Service 和 Client)
#

前言
#

在本篇文章中，我們將實作一個 ROS2 的巡邏服務節點，透過 Service 和 Topic 的結合來控制 turtlesim 中的烏龜移動到指定位置。這個專案展示了如何在 ROS2 中實現服務端（Server）和客戶端（Client）的通信，以及基本的運動控制。

專案架構
#

1. 檔案結構
# 

patrol/
├── CMakeLists.txt          # 構建配置文件
├── package.xml             # 包配置文件
├── src/
│   ├── service.cpp         # 服務端節點實現
│   └── client.cpp          # 客戶端節點實現
└── srv/
    └── Patrol.srv          # 服務介面定義

2. 服務介面定義
# 

# Patrol.srv
float64 target_x    # 目標 X 座標
float64 target_y    # 目標 Y 座標
---
int8 SUCCESS = 1    # 成功狀態碼
int8 FAIL = 0      # 失敗狀態碼
int8 result        # 執行結果

3. 完整程式碼實現
#

3.1 服務端程式碼 (service.cpp)
# 

// 包含必要的標頭文件
#include "rclcpp/rclcpp.hpp"              // ROS2 C++ 客戶端庫
#include "patrol/srv/patrol.hpp"          // 生成的服務介面
#include "geometry_msgs/msg/pose.hpp"     // 位置訊息
#include "turtlesim/msg/pose.hpp"         // turtlesim 位置訊息
#include "geometry_msgs/msg/twist.hpp"    // 速度訊息

// 巡邏服務節點類別，繼承自 rclcpp::Node
class PatrolService: public rclcpp::Node {
    private:
        // 成員變量
        double targetX_ {1.0};            // 目標 X 座標，預設值為 1.0
        double targetY_ {1.0};            // 目標 Y 座標，預設值為 1.0
        
        // ROS2 通信組件
        rclcpp::Service<patrol::srv::Patrol>::SharedPtr service_;    // 巡邏服務
        turtlesim::msg::Pose::SharedPtr turtlePose_;                // 儲存烏龜當前位置
        rclcpp::Subscription<turtlesim::msg::Pose>::SharedPtr turtlePoseSub_;  // 訂閱烏龜位置
        rclcpp::Publisher<geometry_msgs::msg::Twist>::SharedPtr publisher_;     // 發布速度命令

    public:
        // 建構函數，初始化節點
        PatrolService(std::string nodeName);
        // 解構函數，使用預設實作
        ~PatrolService() = default;
        
        // 處理巡邏服務請求的回調函數
        void handle_patrol(const std::shared_ptr<patrol::srv::Patrol::Request> request,
                         const std::shared_ptr<patrol::srv::Patrol::Response> response);
        
        // 控制烏龜移動到目標位置的函數
        void move_to_target(double targetX, double targetY);
};

// 建構函數實現
PatrolService::PatrolService(std::string nodeName): Node(nodeName) {
    // 創建位置訂閱者，訂閱烏龜的位置資訊
    turtlePoseSub_ = this->create_subscription<turtlesim::msg::Pose>(
        "/turtle1/pose",    // 話題名稱
        10,                 // QoS 設定
        [this](const turtlesim::msg::Pose::SharedPtr msg) {  // 回調函數
            turtlePose_ = msg;  // 更新烏龜位置
            move_to_target(turtlePose_->x, turtlePose_->y);  // 執行移動控制
        }
    );

    // 創建速度發布者，用於發送移動命令
    publisher_ = this->create_publisher<geometry_msgs::msg::Twist>("/turtle1/cmd_vel", 10);

    // 創建巡邏服務
    service_ = this->create_service<patrol::srv::Patrol>(
        "patrol",  // 服務名稱
        std::bind(&PatrolService::handle_patrol, this, std::placeholders::_1, std::placeholders::_2)  // 服務回調函數
    );
}

// 服務回調函數實現
void PatrolService::handle_patrol(
    const std::shared_ptr<patrol::srv::Patrol::Request> request,
    const std::shared_ptr<patrol::srv::Patrol::Response> response) 
{
    // 檢查目標座標是否在有效範圍內（0-12之間）
    if((request->target_x > 0.0f && request->target_x < 12.0f) && 
       (request->target_y > 0.0f && request->target_y < 12.0f)) 
    {
        // 更新目標位置
        targetX_ = request->target_x;
        targetY_ = request->target_y;
        response->result = patrol::srv::Patrol::Response::SUCCESS;  // 設置成功狀態
    } else {
        // 座標無效，設置失敗狀態
        response->result = patrol::srv::Patrol::Response::FAIL;
    }
}

// 移動控制函數實現
void PatrolService::move_to_target(double currentX, double currentY) {
    // 創建速度訊息物件
    auto twist = geometry_msgs::msg::Twist();
    
    // 計算當前位置到目標位置的差值
    double dx = targetX_ - currentX;
    double dy = targetY_ - currentY;
    // 計算距離
    double distance = std::sqrt(dx * dx + dy * dy);
    // 計算目標角度（使用 atan2 確保角度在正確的象限）
    double target_angle = std::atan2(dy, dx);

    // 計算角度差
    double angle_diff = target_angle - turtlePose_->theta;
    
    // 如果距離大於閾值（0.1），則繼續移動
    if (distance > 0.1) {
        // 如果角度差大於閾值，優先調整方向
        if(fabs(angle_diff) > 0.2) {
            twist.angular.z = 1.0 * angle_diff;  // 角速度與角度差成正比
        } else {
            twist.linear.x = 0.5 * distance;     // 線速度與距離成正比
        }

        // 輸出當前位置資訊，用於除錯
        RCLCPP_INFO(this->get_logger(), "currentX: %f, currentY: %f", currentX, currentY);
        
        // 發布速度命令
        publisher_->publish(twist);
    }
}

// 主函數
int main(int argc, char **argv) {
    rclcpp::init(argc, argv);     // 初始化 ROS2
    auto node = std::make_shared<PatrolService>("patrol_service_node");  // 創建服務節點
    rclcpp::spin(node);           // 運行節點
    rclcpp::shutdown();           // 關閉 ROS2
    return 0;
}

3.2 客戶端程式碼 (client.cpp)
# 

#include "rclcpp/rclcpp.hpp"              // ROS2 C++ 客戶端庫
#include "patrol/srv/patrol.hpp"          // 生成的服務介面
#include "rclcpp/time.hpp"                // 時間相關功能
#include <random>                         // 隨機數生成

// 巡邏客戶端類別，繼承自 rclcpp::Node
class Client: public rclcpp::Node {
private:
    // ROS2 通信組件
    rclcpp::Client<patrol::srv::Patrol>::SharedPtr client_;         // 服務客戶端
    std::shared_ptr<patrol::srv::Patrol::Request> request_;         // 請求物件
    rclcpp::TimerBase::SharedPtr timer_;                           // 定時器

public: 
    // 建構函數，初始化節點
    Client(std::string nodeName): Node(nodeName) {
        // 創建服務客戶端
        this->client_ = this->create_client<patrol::srv::Patrol>("patrol");
        RCLCPP_INFO(this->get_logger(), "Client has been created");
        
        // 創建定時器，每5秒發送一次請求
        this->timer_ = this->create_wall_timer(
            std::chrono::seconds(5), 
            std::bind(&Client::cb, this)
        );
    }
    
    // 解構函數，使用預設實作
    ~Client() = default;
    
    // 定時器回調函數
    void cb() {
        // 等待服務可用
        while(!this->client_->wait_for_service(std::chrono::seconds(1))) {
            if(!rclcpp::ok()){
                RCLCPP_ERROR(this->get_logger(), "Interrupted while waiting for the service. Exiting.");
                return;
            }
            RCLCPP_INFO(this->get_logger(), "Service not available, waiting again...");
        }
        
        // 創建請求物件
        this->request_ = std::make_shared<patrol::srv::Patrol::Request>();

        // 生成隨機目標位置
        std::random_device rd;
        std::mt19937 gen(rd());
        std::uniform_real_distribution<double> dis_float(1.0, 10.0);

        // 設置隨機目標位置
        this->request_->target_x = dis_float(gen);
        this->request_->target_y = dis_float(gen);
        
        // 輸出請求資訊
        RCLCPP_INFO(this->get_logger(), "Requesting patrol to (%f, %f)", 
                    this->request_->target_x, this->request_->target_y);
        
        // 發送非同步請求
        auto result = this->client_->async_send_request(
            this->request_,
            [this](rclcpp::Client<patrol::srv::Patrol>::SharedFuture future) {
                auto response = future.get();
                if(response->result == patrol::srv::Patrol::Response::SUCCESS) {
                    RCLCPP_INFO(this->get_logger(), "Patrol to (%f, %f) successful", 
                               this->request_->target_x, this->request_->target_y);
                } else {
                    RCLCPP_ERROR(this->get_logger(), "Patrol to (%f, %f) failed", 
                                this->request_->target_x, this->request_->target_y);
                }
            }
        );
    }
};

// 主函數
int main(int argc, char **argv) {
    rclcpp::init(argc, argv);     // 初始化 ROS2
    auto client = std::make_shared<Client>("patrol_client_node");  // 創建客戶端節點
    rclcpp::spin(client);         // 運行節點
    rclcpp::shutdown();           // 關閉 ROS2
    return 0;
}

4. 配置文件說明
#

4.1 CMakeLists.txt
# 

# 指定 CMake 最低版本要求
cmake_minimum_required(VERSION 3.8)
# 定義專案名稱
project(patrol)

# 如果使用 GCC 或 Clang 編譯器，添加編譯警告選項
if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# 尋找並載入所需的依賴套件
find_package(ament_cmake REQUIRED)              # ROS 2 CMake 工具
find_package(rclcpp REQUIRED)                  # ROS 2 C++ 客戶端程式庫
find_package(std_msgs REQUIRED)                # 標準訊息類型
find_package(rosidl_default_generators REQUIRED)# 服務介面生成器
find_package(turtlesim REQUIRED)               # 烏龜模擬器套件
find_package(geometry_msgs REQUIRED)           # 幾何訊息類型

# 生成服務介面
rosidl_generate_interfaces(${PROJECT_NAME}
  "srv/Patrol.srv"
)

# 創建可執行檔
add_executable(service_node src/service.cpp)

# 為可執行檔添加依賴
ament_target_dependencies(service_node
  rclcpp        # ROS 2 C++ 客戶端程式庫
  std_msgs      # 標準訊息類型
  geometry_msgs # 幾何訊息類型
  turtlesim     # 烏龜模擬器套件
)

# 安裝可執行檔到指定目錄
install(TARGETS service_node
  DESTINATION lib/${PROJECT_NAME}
)

# 完成 ament 套件的配置
ament_package()

執行方式
#

  1. 編譯專案
cd ros2_ws
colcon build --packages-select patrol
  1. 執行節點
# 終端機 1：啟動 turtlesim
ros2 run turtlesim turtlesim_node

# 終端機 2：啟動巡邏服務端
ros2 run patrol service_node

# 終端機 3：啟動巡邏客戶端
ros2 run patrol client_node
