🐢 ROS2 巡邏服務節點開發教學(使用 Service)
#

前言
#

在本篇文章中，我們將實作一個 ROS2 的巡邏服務節點，透過 Service 和 Topic 的結合來控制 turtlesim 中的烏龜移動到指定位置。這個專案展示了如何在 ROS2 中實現服務端和客戶端的通信，以及基本的運動控制。

專案架構
#

1. 檔案結構
# 

patrol/
├── CMakeLists.txt          # 構建配置文件
├── package.xml             # 包配置文件
├── src/
│   └── service.cpp         # 服務節點實現
└── srv/
    └── Patrol.srv          # 服務介面定義

2. 服務介面定義
# 

# Patrol.srv
float64 target_x    # 目標 X 座標
float64 target_y    # 目標 Y 座標
---
int8 SUCCESS = 1    # 成功狀態碼
int8 FAIL = 0      # 失敗狀態碼
int8 result        # 執行結果

3. 完整程式碼實現
# 

// 包含必要的標頭文件
#include "rclcpp/rclcpp.hpp"              // ROS2 C++ 客戶端庫
#include "patrol/srv/patrol.hpp"          // 生成的服務介面
#include "geometry_msgs/msg/pose.hpp"     // 位置訊息
#include "turtlesim/msg/pose.hpp"         // turtlesim 位置訊息
#include "geometry_msgs/msg/twist.hpp"    // 速度訊息

// 巡邏服務節點類別，繼承自 rclcpp::Node
class PatrolService: public rclcpp::Node {
    private:
        // 成員變量
        double targetX_ {1.0};            // 目標 X 座標
        double targetY_ {1.0};            // 目標 Y 座標
        
        // ROS2 通信組件
        rclcpp::Service<patrol::srv::Patrol>::SharedPtr service_;    // 服務
        turtlesim::msg::Pose::SharedPtr turtlePose_;                // 當前位置
        rclcpp::Subscription<turtlesim::msg::Pose>::SharedPtr turtlePoseSub_;  // 位置訂閱者
        rclcpp::Publisher<geometry_msgs::msg::Twist>::SharedPtr publisher_;     // 速度發布者

    public:
        // 建構函數
        PatrolService(std::string nodeName);
        // 解構函數
        ~PatrolService() = default;
        
        // 服務回調函數
        void handle_patrol(const std::shared_ptr<patrol::srv::Patrol::Request> request,
                         const std::shared_ptr<patrol::srv::Patrol::Response> response);
        
        // 移動控制函數
        void move_to_target(double targetX, double targetY);
};

// 建構函數實現
PatrolService::PatrolService(std::string nodeName): Node(nodeName) {
    // 創建位置訂閱者
    turtlePoseSub_ = this->create_subscription<turtlesim::msg::Pose>(
        "/turtle1/pose", 
        10, 
        [this](const turtlesim::msg::Pose::SharedPtr msg) {
            turtlePose_ = msg;
            move_to_target(turtlePose_->x, turtlePose_->y);
        }
    );

    // 創建速度發布者
    publisher_ = this->create_publisher<geometry_msgs::msg::Twist>("/turtle1/cmd_vel", 10);

    // 創建服務
    service_ = this->create_service<patrol::srv::Patrol>(
        "patrol",
        std::bind(&PatrolService::handle_patrol, this, std::placeholders::_1, std::placeholders::_2)
    );
}

// 服務回調函數實現
void PatrolService::handle_patrol(
    const std::shared_ptr<patrol::srv::Patrol::Request> request,
    const std::shared_ptr<patrol::srv::Patrol::Response> response) 
{
    // 檢查目標座標是否在有效範圍內（0-12之間）
    if((request->target_x > 0.0f && request->target_x < 12.0f) && 
       (request->target_y > 0.0f && request->target_y < 12.0f)) 
    {
        // 更新目標位置
        targetX_ = request->target_x;
        targetY_ = request->target_y;
        response->result = patrol::srv::Patrol::Response::SUCCESS;
    } else {
        // 座標無效
        response->result = patrol::srv::Patrol::Response::FAIL;
    }
}

// 移動控制函數實現
void PatrolService::move_to_target(double currentX, double currentY) {
    // 創建速度訊息
    auto twist = geometry_msgs::msg::Twist();
    
    // 計算距離和角度
    double dx = targetX_ - currentX;
    double dy = targetY_ - currentY;
    double distance = std::sqrt(dx * dx + dy * dy);
    double target_angle = std::atan2(dy, dx);

    // 如果距離大於閾值，則移動
    if (distance > 0.1) {
        // 計算角度差
        double angle_diff = target_angle - turtlePose_->theta;
        
        // 角度歸一化
        if (angle_diff > M_PI) angle_diff -= 2 * M_PI;
        if (angle_diff < -M_PI) angle_diff += 2 * M_PI;

        // 設置角速度和線速度
        twist.angular.z = 1.0 * angle_diff;
        twist.linear.x = 0.5 * distance;

        // 輸出當前位置資訊
        RCLCPP_INFO(this->get_logger(), "currentX: %f, currentY: %f", currentX, currentY);
        
        // 發布速度命令
        publisher_->publish(twist);
    }
}

// 主函數
int main(int argc, char **argv) {
    // 初始化 ROS2
    rclcpp::init(argc, argv);
    // 創建服務節點
    auto node = std::make_shared<PatrolService>("patrol_service_node");
    // 運行節點
    rclcpp::spin(node);
    // 關閉 ROS2
    rclcpp::shutdown();
    return 0;
}

4. 配置文件說明
#

4.1 CMakeLists.txt
# 

# 指定 CMake 最低版本要求
cmake_minimum_required(VERSION 3.8)
# 定義專案名稱
project(patrol)

# 如果使用 GCC 或 Clang 編譯器，添加編譯警告選項
if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# 尋找並載入所需的依賴套件
find_package(ament_cmake REQUIRED)              # ROS 2 CMake 工具
find_package(rclcpp REQUIRED)                  # ROS 2 C++ 客戶端程式庫
find_package(std_msgs REQUIRED)                # 標準訊息類型
find_package(rosidl_default_generators REQUIRED)# 服務介面生成器
find_package(turtlesim REQUIRED)               # 烏龜模擬器套件
find_package(geometry_msgs REQUIRED)           # 幾何訊息類型

# 生成服務介面
rosidl_generate_interfaces(${PROJECT_NAME}
  "srv/Patrol.srv"
)

# 創建可執行檔
add_executable(service_node src/service.cpp)

# 為可執行檔添加依賴
ament_target_dependencies(service_node
  rclcpp        # ROS 2 C++ 客戶端程式庫
  std_msgs      # 標準訊息類型
  geometry_msgs # 幾何訊息類型
  turtlesim     # 烏龜模擬器套件
)

# 安裝可執行檔到指定目錄
install(TARGETS service_node
  DESTINATION lib/${PROJECT_NAME}
)

# 完成 ament 套件的配置
ament_package()

執行方式
#

  1. 編譯專案
cd ros2_ws
colcon build --packages-select patrol
  1. 執行節點
# 終端機 1：啟動 turtlesim
ros2 run turtlesim turtlesim_node

# 終端機 2：啟動巡邏服務
ros2 run patrol patrol_service

# 終端機 3：發送服務請求
ros2 service call /patrol patrol/srv/Patrol "{target_x: 5.0, target_y: 5.0}"

使用 RQT 測試
#

RQT 是 ROS 的圖形化工具，可以用來監控和測試節點。以下是使用 RQT 測試巡邏服務的步驟：

  1. 啟動 RQT
rqt

  1. 使用 Service Caller

    • 在菜單中選擇 Plugins -> Services -> Service Caller
    • 在 Service 下拉選單中選擇 /patrol
    • 填入測試數據： 
      {
  "target_x": 5.0,
  "target_y": 5.0
}
    • 點擊 “Call” 按鈕發送請求

  2. 使用 Plot 工具監控位置

    • 選擇 Plugins -> Visualization -> Plot
    • 添加以下話題進行監控：
      • /turtle1/pose/x：烏龜的 X 座標
      • /turtle1/pose/y：烏龜的 Y 座標
      • /turtle1/pose/theta：烏龜的角度

  3. 使用 Topic Monitor

    • 選擇 Plugins -> Topics -> Topic Monitor
    • 可以監控以下話題：
      • /turtle1/cmd_vel：發布的速度命令
      • /turtle1/pose：烏龜的當前位置

技術要點說明
#

  1. 服務通信機制

    • 使用 Service 處理一次性請求
    • 通過 Topic 實現連續的位置和速度更新
    • 結合同步（Service）和異步（Topic）通信

  2. 運動控制

    • 使用比例控制實現平滑移動
    • 考慮角度連續性，避免突變
    • 距離和角度的實時計算

  3. 錯誤處理

    • 檢查座標範圍有效性
    • 回傳適當的執行結果
    • 移動閾值控制

改進建議
#

  1. PID 控制優化
// 添加積分和微分項
class PatrolService {
private:
    double angle_integral = 0.0;
    double last_angle_error = 0.0;
    
    void move_to_target() {
        double angle_error = target_angle - current_angle;
        angle_integral += angle_error;
        double angle_derivative = angle_error - last_angle_error;
        
        twist.angular.z = Kp * angle_error + 
                         Ki * angle_integral + 
                         Kd * angle_derivative;
        
        last_angle_error = angle_error;
    }
};
  1. 超時處理
class PatrolService {
private:
    rclcpp::Time start_time_;
    const double timeout_duration_ = 10.0;  // 10 秒超時
    
    bool move_timeout() {
        return (this->now() - start_time_).seconds() > timeout_duration_;
    }
    
    void handle_patrol() {
        start_time_ = this->now();
        if (move_timeout()) {
            response->result = patrol::srv::Patrol::Response::FAIL;
            return;
        }
    }
};
