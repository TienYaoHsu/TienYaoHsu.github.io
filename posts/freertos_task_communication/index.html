🤝 FreeRTOS Task 通訊方法
#

1. Queue 通訊機制
#

1.1 基本概念
#

Queue 是 FreeRTOS 中最常用的通訊機制，適合用於傳送資料。

1.2 程式範例
# 

#include "FreeRTOS.h"
#include "task.h"
#include "queue.h"
#include "xil_printf.h" // For xil_printf

QueueHandle_t dataQueue;

void vSenderTask(void *pvParameters) {
    int32_t valueToSend = 0;
    
    while(1) {
        xQueueSend(dataQueue, &valueToSend, portMAX_DELAY);
        valueToSend++;
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

void vReceiverTask(void *pvParameters) {
    int32_t receivedValue;
    
    while(1) {
        if(xQueueReceive(dataQueue, &receivedValue, portMAX_DELAY) == pdPASS) {
            xil_printf("接收到的值: %d\r\n", receivedValue);
        }
    }
}

int main(void) {
    dataQueue = xQueueCreate(10, sizeof(int32_t));
    if(dataQueue != NULL) {
        xTaskCreate(vSenderTask, "Sender", configMINIMAL_STACK_SIZE, NULL, 1, NULL);
        xTaskCreate(vReceiverTask, "Receiver", configMINIMAL_STACK_SIZE, NULL, 1, NULL);
        vTaskStartScheduler();
    }
    while(1);
}

2. Semaphore 通訊機制
#

2.1 Binary Semaphore
#

適用於簡單的同步和事件通知。

#include "FreeRTOS.h"
#include "task.h"
#include "semphr.h"
#include "xil_printf.h" // For xil_printf

SemaphoreHandle_t binarySemaphore;

void vTask1(void *pvParameters) {
    while(1) {
        xil_printf("Task1 完成操作\r\n");
        xSemaphoreGive(binarySemaphore);
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

void vTask2(void *pvParameters) {
    while(1) {
        if(xSemaphoreTake(binarySemaphore, portMAX_DELAY) == pdTRUE) {
            xil_printf("Task2 收到通知並執行\r\n");
        }
    }
}

int main(void) {
    binarySemaphore = xSemaphoreCreateBinary();
    if(binarySemaphore != NULL) {
        xTaskCreate(vTask1, "Task1", configMINIMAL_STACK_SIZE, NULL, 1, NULL);
        xTaskCreate(vTask2, "Task2", configMINIMAL_STACK_SIZE, NULL, 1, NULL);
        vTaskStartScheduler();
    }
    while(1);
}

3. Mutex 互斥機制
#

3.1 保護共享資源
#

Mutex 主要用於保護共享資源，避免多個 task 同時存取。

#include "FreeRTOS.h"
#include "task.h"
#include "semphr.h"
#include "xil_printf.h" // For xil_printf

SemaphoreHandle_t mutexHandle;
int shared_resource = 0;

void vTask1(void *pvParameters) {
    while(1) {
        if(xSemaphoreTake(mutexHandle, portMAX_DELAY) == pdTRUE) {
            shared_resource++;
            xil_printf("Task1: shared_resource = %d\r\n", shared_resource);
            xSemaphoreGive(mutexHandle);
        }
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

int main(void) {
    mutexHandle = xSemaphoreCreateMutex();
    if(mutexHandle != NULL) {
        xTaskCreate(vTask1, "Task1", configMINIMAL_STACK_SIZE, NULL, 1, NULL);
        vTaskStartScheduler();
    }
    while(1);
}

4. Event Groups 事件組
#

4.1 多事件同步
#

適合需要等待多個條件同時成立的場景。

#include "FreeRTOS.h"
#include "task.h"
#include "event_groups.h"
#include "xil_printf.h" // For xil_printf

#define EVENT_BIT_0 (1 << 0)
#define EVENT_BIT_1 (1 << 1)
#define ALL_EVENTS (EVENT_BIT_0 | EVENT_BIT_1)

EventGroupHandle_t eventGroup;

void vTask3(void *pvParameters) {
    EventBits_t bits;
    while(1) {
        bits = xEventGroupWaitBits(eventGroup,
                                 ALL_EVENTS,    
                                 pdTRUE,        
                                 pdTRUE,        
                                 portMAX_DELAY);
        
        if((bits & ALL_EVENTS) == ALL_EVENTS) {
            xil_printf("Task3 收到所有事件\r\n");
        }
    }
}

int main(void) {
    eventGroup = xEventGroupCreate();
    if(eventGroup != NULL) {
        xTaskCreate(vTask3, "Task3", configMINIMAL_STACK_SIZE, NULL, 1, NULL);
        vTaskStartScheduler();
    }
    while(1);
}

5. Task Notification
#

5.1 輕量級通訊
#

最新且效能最佳的通訊方式。

#include "FreeRTOS.h"
#include "task.h"
#include "xil_printf.h" // For xil_printf

TaskHandle_t xTask2Handle = NULL;

void vTask1(void *pvParameters) {
    while(1) {
        xTaskNotify(xTask2Handle, 0x01, eSetBits);
        xil_printf("Task1 發送通知\r\n");
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

void vTask2(void *pvParameters) {
    uint32_t ulNotificationValue;
    while(1) {
        if(xTaskNotifyWait(0x00, 0xFFFFFFFF, &ulNotificationValue, portMAX_DELAY) == pdTRUE) {
            xil_printf("Task2 收到通知\r\n");
        }
    }
}

int main(void) {
    xTaskCreate(vTask1, "Task1", configMINIMAL_STACK_SIZE, NULL, 1, NULL);
    xTaskCreate(vTask2, "Task2", configMINIMAL_STACK_SIZE, NULL, 1, &xTask2Handle);
    vTaskStartScheduler();
    while(1);
}

6. 使用建議
#

6.1 選擇合適的通訊方式
#

  • 簡單通訊：優先使用 Task Notification
  • 資料傳輸：使用 Queue
  • 資源保護：使用 Mutex
  • 多事件同步：使用 Event Groups

6.2 效能考量
#

  1. Task Notification 效能最好
  2. Queue 適合資料傳輸但開銷較大
  3. Semaphore 和 Mutex 適合同步場景

💡 提示：

  • 在使用這些機制時，要注意避免死鎖
  • 正確設定任務優先級
  • 謹慎使用阻塞時間
  • 建議在開發初期就規劃好通訊架構
