        this->declare_parameter("Target_Y", 1.0, descriptor);

            std::bind(&ServiceParameters::parametersCallback, this, std::placeholders::_1));

        turtlePoseSub_ = this->create_subscription<turtlesim::msg::Pose>(

            "patrol",
            std::bind(&ServiceParameters::handle_patrol, this, std::placeholders::_1, std::placeholders::_2)
        );

                targetX_ = param.as_double();
                RCLCPP_INFO(this->get_logger(), "Target_X updated to: %f", targetX_);
            } else if(param.get_name() == "Target_Y") {
                targetY_ = param.as_double();
                RCLCPP_INFO(this->get_logger(), "Target_Y updated to: %f", targetY_);
            }
        }
        
        return result;
    }

        const std::shared_ptr<patrol::srv::Patrol::Request> request,
        const std::shared_ptr<patrol::srv::Patrol::Response> response) 
    {
        if((request->target_x > 0.0f && request->target_x < 12.0f) && 
           (request->target_y > 0.0f && request->target_y < 12.0f)) 
        {
            targetX_ = request->target_x;
            targetY_ = request->target_y;
            response->result = patrol::srv::Patrol::Response::SUCCESS;
        } else {
            response->result = patrol::srv::Patrol::Response::FAIL;
        }
    }

    // ç§»å‹•æ§åˆ¶å‡½æ•¸

        auto twist = geometry_msgs::msg::Twist();
        
        double dx = targetX_ - currentX;
        double dy = targetY_ - currentY;
        double distance = std::sqrt(dx * dx + dy * dy);
        double target_angle = std::atan2(dy, dx);
        double angle_diff = target_angle - turtlePose_->theta;
        
        if (distance > 0.1) {
            if(fabs(angle_diff) > 0.2) {
                twist.angular.z = 1.0 * angle_diff;
            } else {
                twist.linear.x = 0.5 * distance;
            }
            
            RCLCPP_INFO(this->get_logger(), "Moving to target: (%f, %f), current: (%f, %f)", 
                       targetX_, targetY_, currentX, currentY);
            publisher_->publish(twist);
        }
    }
};
